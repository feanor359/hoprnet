pub enum IdentityRetrievalModes<'a> {
    FromFile {
        password: &'a str,
        id_path: &'a str,
    },
    FromPrivateKey {
        private_key: &'a str,
    },
    FromIdIntoFile {
        id: Uuid,
        password: &'a str,
        id_path: &'a str,
    },
